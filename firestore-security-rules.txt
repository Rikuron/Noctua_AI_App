rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Notebooks collection
    match /notebooks/{notebookId} {
      // Helper function to check if user owns this notebook
      function isOwner() {
        return request.auth != null && 
          resource.data.userId == request.auth.uid;
      }
      
      // Allow read if user owns the notebook
      allow read: if isOwner();
      
      // Allow create if user is authenticated and sets themselves as owner
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      // Allow update/delete if user owns the notebook
      allow update, delete: if isOwner();
      
      // Sources subcollection
      match /sources/{sourceId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      // Summaries subcollection
      match /summaries/{summaryId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      // Presentations subcollection
      match /presentations/{presentationId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      // Chats subcollection
      match /chats/{chatId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
    }
    
    // Public sources collection (repository)
    match /public-sources/{sourceId} {
      allow read: if true; // Anyone can read public sources
      allow write: if request.auth != null; // Only authenticated users can write
    }
    
    // Deny all other documents
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
