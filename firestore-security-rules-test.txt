rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // TEMPORARY TEST RULES - More permissive for debugging
    // Replace with secure rules after testing
    
    // Notebooks collection
    match /notebooks/{notebookId} {
      // Allow read if user owns the notebook
      allow read: if request.auth != null && 
        resource.data.userId == request.auth.uid;
      
      // Allow create if user is authenticated and sets themselves as owner
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      // Allow update/delete if user owns the notebook
      allow update, delete: if request.auth != null && 
        resource.data.userId == request.auth.uid;
      
      // Subcollections - More permissive for testing
      // Check if notebook exists and user owns it
      match /sources/{sourceId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      match /summaries/{summaryId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      match /presentations/{presentationId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
      
      match /chats/{chatId} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/notebooks/$(notebookId)) &&
          get(/databases/$(database)/documents/notebooks/$(notebookId)).data.userId == request.auth.uid;
      }
    }
    
    // Public sources collection
    match /public-sources/{sourceId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Deny all other documents
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
